def save() # CREATE
  db = PG.connect({dbname: 'property_tracker', host: 'localhost'})
  sql = "INSERT INTO properties
  (
    address,
    value,
    bedrooms,
    build
  )
  VALUES
  (
    $1,$2,$3,$4
  )
  RETURNING id"
  values = [@address, @value, @bedrooms, @build]
  db.prepare("save", sql)
  @id = db.exec_prepared("save", values)[0]["id"].to_i
  db.close()
end
